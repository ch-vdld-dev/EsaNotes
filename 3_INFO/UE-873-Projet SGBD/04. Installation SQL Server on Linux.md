### Docker-compose

```yaml
version: '3.3'

services:
  sql-server:
    container_name: sql-docker
    image: mcr.microsoft.com/mssql/server:2022-latest
    # restart: always
    environment:
      ACCEPT_EULA: "Y"
      SA_PASSWORD: "<your password>"
    ports:
      - 1433:1433
    volumes:
      - my-volume:/home/GuideCharon/Documents/Cours/SGBD/mssql
      
volumes:
  my-volume:
```
